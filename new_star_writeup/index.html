    def dec(cipher,i,li):
    if(cipher[i]>='a'and cipher[i]<='z'):
        return chr((ord(cipher[i])+((-li[i])%26)-ord('a'))%26+ord('a'))
    elif(cipher[i]>='A'and cipher[i]<='Z'):
        return chr((ord(cipher[i])+((-li[i])%26)-ord('A'))%26+ord('A'))
    else :
        return cipher[i]
